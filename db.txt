create schema 340_progetto;
use 340_progetto;

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(100) NOT NULL,
    role ENUM('user','admin') NOT NULL
);

CREATE TABLE vm_requests (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    vm_type ENUM('bronze','silver','gold') NOT NULL,
    status ENUM('pending','approved','created','rejected') DEFAULT 'pending',
    created_at DATETIME,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE vm_instances (
    id INT AUTO_INCREMENT PRIMARY KEY,
    request_id INT NOT NULL,
    hostname VARCHAR(100),
    ip_address VARCHAR(50),
    vm_user VARCHAR(50),
    vm_password VARCHAR(50),
    cores INT,
    memory INT,     
    disk INT,    
    FOREIGN KEY (request_id) REFERENCES vm_requests(id) ON DELETE CASCADE
);
